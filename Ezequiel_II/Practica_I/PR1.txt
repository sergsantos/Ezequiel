CREATE DATABASE PRAC1
GO

CREATE TABLE ROL_PERSONA
(
P_ID_PERSONA      VARCHAR(20) FOREIGN KEY REFERENCES PERSONA,
R_ID_ROL          VARCHAR(20) FOREIGN KEY REFERENCES ROL,
FECHA             DATE
)
GO
CREATE TABLE EVENTOS
GO
(
ID_EVENTO         VARCHAR(20) PRIMARY KEY NOT NULL,
NOMBRE_EVENTO     VARCHAR(50),
TIPO_EVENTO       VARCHAR(50),
FECHA_EVENTO      DATE,
DESCRIPCION       VARCHAR(50)
)
GO
CREATE TABLE ESTATUS
(
ID_ESTATUS         VARCHAR(20) PRIMARY KEY NOT NULL,
E_ID_EVENTO        VARCHAR(20) FOREIGN KEY REFERENCES EVENTOS
)
GO
CREATE TABLE CASOS
(
ID_CASOS           VARCHAR(20) PRIMARY KEY NOT NULL,
NOMBRE_CASO        VARCHAR(50),
E_ID_EVENTO        VARCHAR(20) FOREIGN KEY REFERENCES EVENTOS,
D_ID_DEPARTAMENTOS VARCHAR(20) FOREIGN KEY REFERENCES DEPARTAMENTOS,
DESCRIPCION        VARCHAR(50),
FECHA_CASO         DATE
)
CREATE TABLE DEPARTAMENTO
GO
(
ID_DEPARTAMENTO   VARCHAR(20) PRIMARY KEY NOT NULL,
NOMBRE_DEPTO      VARCHAR(50) NOT NULL,
DESCRIPCION       VARCHAR(50),
)
GO
CREATE TABLE PERSONA
(
ID_PERSONA        VARCHAR(20) PRIMARY KEY NOT NULL,
NOMBRE            VARCHAR(50) NOT NULL,
APATERNO          VARCHAR(50) NOT NULL,
AMATERNO          VARCHAR(50) NOT NULL,
FECHANACIMIENTO   DATE NOT NULL,
DIRECCION         VARCHAR(50) NULL,
E_MAIL            VARCHAR(50) NULL,
TELEFONO          VARCHAR(20) NULL,
)
GO
CREATE TABLE ROL
(
ID_ROL             VARCHAR(20) PRIMARY KEY NOT NULL,
TIPO               VARCHAR(20) NULL,
)
GO
CREATE TABLE PERSONA_CASOS
(
C_ID_CASOS       VARCHAR(20) FOREIGN KEY REFERENCES CASOS,
P_ID_PERSONA     VARCHAR(20) FOREIGN KEY REFERENCES PERSONA
)
